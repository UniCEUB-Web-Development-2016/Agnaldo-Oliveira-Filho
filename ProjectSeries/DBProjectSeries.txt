
Table	Create Table

episode	CREATE TABLE `episode` (
 	`cod_episode` int(11) NOT NULL AUTO_INCREMENT,
 	`series_name` varchar(30) NOT NULL,
	 `season` int(11) NOT NULL,
 	PRIMARY KEY (`cod_episode`))


lista	CREATE TABLE `lista` (
 	`idt_lista` int(11) NOT NULL AUTO_INCREMENT,
 	`series_name` varchar(50) NOT NULL,
 	`season` int(11) NOT NULL,
 	`episode` varchar(50) NOT NULL,
 	`position` int(11) NOT NULL,
 	PRIMARY KEY (`idt_lista`))


series	CREATE TABLE `series` (
 	`cod_series` int(11) NOT NULL AUTO_INCREMENT,
 	`series_name` varchar(30) NOT NULL,
 	`initials` varchar(15) NOT NULL,
 	`category` varchar(20) NOT NULL,
 	`year_series` int(11) NOT NULL,
 	`director` varchar(50) NOT NULL,
 	PRIMARY KEY (`cod_series`))


user	CREATE TABLE `user` (
 	`idt_user` int(11) NOT NULL AUTO_INCREMENT,
 	`firstName` varchar(20) NOT NULL,
 	`lastName` varchar(50) NOT NULL,
 	`email` varchar(100) NOT NULL,
 	`birthDate` date NOT NULL,
 	`city` varchar(50) NOT NULL,
 	`state` varchar(30) NOT NULL,
 	`pass` varchar(50) NOT NULL,
 	PRIMARY KEY (`idt_user`))
